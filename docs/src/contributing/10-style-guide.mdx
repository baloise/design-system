## Imports

Always import from the `@utils` and `@global` aliases â€” never from relative paths. Merge imports from the same source into one line.

```ts
// âťŚ Don't
import { Loggable, Logger } from '../../utils/log'
import { inheritAttributes } from '../../utils/attributes'
import { DsConfigState } from '../../global'

// âś… Do
import { Loggable, Logger, type LogInstance, inheritAttributes } from '@utils'
import { DsConfigState } from '@global'
```

## Component Tag and Class Name

The HTML tag must start with `ds-`. The class name must **not** have a prefix.

```ts
// âťŚ Don't
@Component({ tag: 'button' }) export class DsButton {}

// âś… Do
@Component({ tag: 'ds-button' }) export class Button {}
```

---

## Interfaces â€” Const Arrays and Derived Types

Define allowed values as a `const` array in the `.interfaces.ts` file. Then derive the TypeScript type from that array. This keeps values and types in sync automatically.

```ts
// âś… In button.interfaces.ts
namespace DS {
  export const BUTTON_SIZES = ['', 'sm', 'md', 'lg'] as const
  export type ButtonSize = (typeof BUTTON_SIZES)[number]
}
```

```ts
// âś… In button.tsx
@Prop()
@ValidateEmptyOrOneOf(...DS.BUTTON_SIZES)
readonly size: DS.ButtonSize = ''
```

Never duplicate the values by writing them again in the component file.

## Props

### Always use `readonly`

Every `@Prop()` that the component never reassigns must be `readonly`.

```ts
// âťŚ Don't
@Prop() label: string

// âś… Do
@Prop() readonly label: string
```

### Always add a type annotation

When a prop has a default value, always write the type explicitly.

```ts
// âťŚ Don't
@Prop() readonly disabled = false
@Prop() readonly size = 'md'

// âś… Do
@Prop() readonly disabled: boolean = false
@Prop() readonly size: DS.ButtonSize = 'md'
```

### Use `reflect: true` only for state props

State props (like `disabled`, `value`, `invalid`, `checked`, `open`, `loading`, `readonly`, `active`) must reflect to the HTML attribute. Visual props (like `size`, `color`, `label`, `align`) must not.

```ts
// âś… State prop â€” reflects to HTML attribute
@Prop({ reflect: true }) readonly disabled: boolean = false

// âś… Visual prop â€” does not reflect
@Prop() readonly size: DS.ButtonSize = 'md'
```

### Use empty string default for optional enum props

Optional enum props should default to `''` (not `undefined`). The const array must include `''` as the first value. In render logic, check truthiness (`!!this.color`) rather than `!== undefined`.

```ts
// âťŚ Don't
@Prop() readonly color?: ButtonColor

// âś… Do
@Prop() readonly color: ButtonColor = ''

// âś… In interfaces.ts
export const BUTTON_COLORS = ['', 'primary', 'secondary'] as const
```

### Keep `@Prop` and `@Watch` together

If a prop has a watcher, place the `@Watch` directly below the `@Prop`.

```ts
// âś…
@Prop({ reflect: true }) readonly value: string = ''
@Watch('value')
valueChanged() {
  // ...
}
```

### Order props alphabetically

Sort `@Prop()`, `@State()`, and `@Event()` declarations alphabetically within their group.

```ts
// âś…
@Prop() readonly align: string = ''
@Prop() readonly disabled: boolean = false
@Prop() readonly label: string = ''
```

## Prop Validation

Every component must have a `private validateProps()` method â€” even if there are no props. Call it from both `connectedCallback()` and `componentWillUpdate()`.

```ts
// âś…
connectedCallback(): void {
  this.validateProps()
}

componentWillUpdate() {
  this.validateProps()
}

private validateProps() {
  checkEmptyOrType(this, 'label', 'string')
  checkEmptyOrOneOf(this, 'size', DS.BUTTON_SIZES)
}
```

If the component has no props, add a comment:

```ts
private validateProps() {
  // no props to validate
}
```

Available checkers from `@utils`:

<Markdown>
  {`
| Checker | When to use |
| --- | --- |
| \`checkEmptyOrType(this, 'prop', 'string')\` | Optional string, number, or boolean prop |
| \`checkEmptyOrOneOf(this, 'prop', [...values])\` | Optional enum prop |
| \`checkRequiredAndType(this, 'prop', 'string')\` | Required string, number, or boolean prop |
| \`checkRequiredAndOneOf(this, 'prop', [...values])\` | Required enum prop |
  `}
</Markdown>

## Naming Conventions

### `@Listen()` handlers â€” `listenTo*`

```ts
// âťŚ Don't
@Listen('click') onClick() {}

// âś… Do
@Listen('click') listenToClick() {}
```

### `@Watch()` handlers â€” `*Changed`

```ts
// âťŚ Don't
@Watch('value') onValueChange() {}

// âś… Do
@Watch('value') valueChanged() {}
```

### DOM event handlers â€” `handle*` as arrow functions

```ts
// âťŚ Don't
onClick() {}
handleClick() {}

// âś… Do
handleClick = () => {}
handleKeyDown = (event: KeyboardEvent) => {}
```

## Events â€” `ds` Prefix

All custom events must start with `ds`.

```ts
// âťŚ Don't
@Event() change: EventEmitter<string>

// âś… Do
@Event() dsChange: EventEmitter<string>
@Event() dsCloseClick: EventEmitter<void>
```

---

## `@Method()` â€” Always `async`

Public methods must be `async` and return a `Promise`.

```ts
// âťŚ Don't
@Method()
open() {}

// âś… Do
@Method()
async open(): Promise<void> {}
```

---

## Method Visibility â€” `private`

All methods must be `private` except:

- Stencil lifecycle hooks (`connectedCallback`, `componentWillLoad`, etc.)
- `@Method()` decorated methods
- `@Watch()` decorated methods
- `@Listen()` decorated methods
- `render()`

```ts
// âťŚ Don't
computeLabel() { ... }

// âś… Do
private computeLabel() { ... }
```

---

## `ComponentInterface` and `Loggable`

Every component must implement `ComponentInterface` and `Loggable`, and wire up the logger.

```ts
import { Loggable, Logger, type LogInstance } from '@utils'

export class Button implements ComponentInterface, Loggable {
  log!: LogInstance

  @Logger('ds-button')
  createLogger(log: LogInstance) {
    this.log = log
  }
}
```

## JSDoc Comments

Every `@Prop()`, `@Event()`, and `@Method()` must have a JSDoc comment. These comments are used to generate the API documentation automatically.

```ts
// âťŚ Don't
@Prop() readonly label: string

// âś… Do
/** The visible label text displayed inside the button. */
@Prop() readonly label: string
```

---

## Component Description and `@slot` / `@part` Tags

The JSDoc block above `@Component(...)` must have:

1. **One sentence** describing what the component is and what it does.
2. A `@slot` tag for every `<slot />` used in `render()`.
3. A `@part` tag for every `part="..."` attribute used in `render()`.

```ts
/**
 * Button provides a clickable element for triggering actions, submitting forms, or navigating.
 *
 * @slot - Button label text and/or icon children.
 * @part native - The native `<button>` or `<a>` element.
 * @part label - The text label wrapper.
 */
@Component({ tag: 'ds-button', shadow: true })
export class Button implements ComponentInterface {}
```

```ts
 * ------------------------------------------------------

 * ------------------------------------------------------

 * ------------------------------------------------------
 */

/**
 * RENDER
 * ------------------------------------------------------
 */
```

## SCSS â€” CSS Classes, Not Attribute Selectors

Always use CSS class selectors for styling. Never use attribute selectors.

When a prop is set via JavaScript, the HTML attribute may not exist. But CSS classes added in `render()` are always present.

```scss
// âťŚ Don't â€” breaks when prop is set via JavaScript
:host([color="primary"]) { ... }
:host([disabled]) { ... }

// âś… Do â€” always works
:host(.is-primary) { ... }
:host(.is-disabled) { ... }
```

Add the matching classes in `render()`:

```tsx
render() {
  return (
    <Host
      class={{
        'is-primary': this.color === 'primary',
        'is-disabled': this.disabled,
      }}
    >
      <slot />
    </Host>
  )
}
```
